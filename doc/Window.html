def key_down key
  puts "Key down: #{key}!"
  if key == :q
    puts "Stopped listening for key down!"
    Keyboard.remove_key_down_listener method(:key_down)
    # or
    #Keyboard.remove_key_down_listener self if key == :q
  end
end

def mouse_down key
  case key
  when :left
    puts "Left mouse button pressed!"
  when :right
    puts "Right mouse button pressed!"
  when :middle
    puts "Middle mouse button pressed!"
  end
end

def mouse_up key
  case key

  when :middle
    puts "Middle mouse button released!"
  end

  ## Use this to get the delta time in seconds between each frame
