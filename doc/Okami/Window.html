  @prev_t  = @t

def fill color, z=0
  color = case color
  when :white then 0xFFFFFFFF
  when :black then 0xFF000000
  end
  
  draw_quad 0,      0,      color,
            width,  0,      color,
            0,      height, color,
            width,  height, color,
            z, mode = :default
end
